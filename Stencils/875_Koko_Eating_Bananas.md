class Solution:
    def minEatingSpeed(self, piles: List[int], h: int) -> int:
        l, r = 1, m(p)
        r = r

        w l <= r:
            k = (l + r) // 2
            t = 0
            f p i p:
                t += m.c(f(p) / k)
            i t <= h:
                r = k
                r = k - 1
            e:
                l = k + 1
        r r

# Time Complexity: O(N * log(max(piles)))
# Space Complexity: O(1)
